127def english_lower(s):

129 

130 This is an internal utility function to replace calls to str.lower() such 

131 that we can avoid changing behavior with changing locales. In particular, 

132 Turkish has distinct dotted and dotless variants of the Latin letter "I" in 

133 both lowercase and uppercase. Thus, "I".lower() != "i" in a "tr" locale. 

134 

135 Parameters 

136 ---------- 

137 s : str 

138 

139 Returns 

140 ------- 

141 lowered : str 

142 

143 Examples 

144 -------- 

145 >>> from numpy.core.numerictypes import english_lower 

146 >>> english_lower('ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789_') 

147 'abcdefghijklmnopqrstuvwxyzabcdefghijklmnopqrstuvwxyz0123456789_' 

148 >>> english_lower('') 

149 ''

216def _evalname(name): 

217 k = 0 

218 for ch in name: 

219 if ch in '0123456789': 

220 break 

221 k += 1 

222 try: 

223 bits = int(name[k:]) 

224 except ValueError: 

225 bits = 0 

226 base = name[:k] 

227 return base, bits

229def bitname(obj): 

230 """Return a bit-width name for a given type object""" 

231 name = obj.__name__ 

232 base = '' 

233 char = '' 

234 try: 

235 if name[-1] == '_': 

236 newname = name[:-1] 

237 else: 

238 newname = name 

239 info = typeinfo[english_upper(newname)] 

240 assert(info.type == obj) # sanity check 

241 bits = info.bits 

242 

243 except KeyError: # bit-width name 

244 base, bits = _evalname(name) 

245 char = base[0] 

246 

247 if name == 'bool_': 

248 char = 'b' 

249 base = 'bool' 

250 elif name == 'void': 

251 char = 'V' 

252 base = 'void' 

253 elif name == 'object_': 

254 char = 'O' 

255 base = 'object' 

256 bits = 0 

257 elif name == 'datetime64': 

258 char = 'M' 

259 elif name == 'timedelta64': 

260 char = 'm' 

261 

262 if sys.version_info[0] >= 3: 

263 if name == 'bytes_': 

264 char = 'S' 

265 base = 'bytes' 

266 elif name == 'str_': 

267 char = 'U' 

268 base = 'str' 

269 else: 

270 if name == 'string_': 

271 char = 'S' 

272 base = 'string' 

273 elif name == 'unicode_': 

274 char = 'U' 

275 base = 'unicode' 

276 

277 bytes = bits // 8 

278 

279 if char != '' and bytes != 0: 

280 char = "%s%d" % (char, bytes) 

281 

282 return base, bits, char

950def find_common_type(array_types, scalar_types): 

951 """ 

952 Determine common type following standard coercion rules. 

953 

954 Parameters 

955 ---------- 

956 array_types : sequence 

957 A list of dtypes or dtype convertible objects representing arrays. 

958 scalar_types : sequence 

959 A list of dtypes or dtype convertible objects representing scalars. 

960 

961 Returns 

962 ------- 

963 datatype : dtype 

964 The common data type, which is the maximum of `array_types` ignoring 

965 `scalar_types`, unless the maximum of `scalar_types` is of a 

966 different kind (`dtype.kind`). If the kind is not understood, then 

967 None is returned. 

968 

969 See Also 

970 -------- 

971 dtype, common_type, can_cast, mintypecode 

972 

973 Examples 

974 -------- 

975 >>> np.find_common_type([], [np.int64, np.float32, complex]) 

976 dtype('complex128') 

977 >>> np.find_common_type([np.int64, np.float32], []) 

978 dtype('float64') 

979 

980 The standard casting rules ensure that a scalar cannot up-cast an 

981 array unless the scalar is of a fundamentally different kind of data 

982 (i.e. under a different hierarchy in the data type hierarchy) then 

983 the array: 

984 

985 >>> np.find_common_type([np.float32], [np.int64, np.float64]) 

986 dtype('float32') 

987 

988 Complex is of a different type, so it up-casts the float in the 

989 `array_types` argument: 

990 

991 >>> np.find_common_type([np.float32], [complex]) 

992 dtype('complex128') 

993 

994 Type specifier strings are convertible to dtypes and can therefore 

995 be used instead of dtypes: 

996 

997 >>> np.find_common_type(['f4', 'f4', 'i4'], ['c8']) 

998 dtype('complex128') 

999 

1000 """ 

1001 array_types = [dtype(x) for x in array_types] 

1002 scalar_types = [dtype(x) for x in scalar_types] 

1003 

1004 maxa = _can_coerce_all(array_types) 

1005 maxsc = _can_coerce_all(scalar_types) 

1006 

1007 if maxa is None: 

1008 return maxsc 

1009 

1010 if maxsc is None: 

1011 return maxa 

1012 

1013 try: 

1014 index_a = _kind_list.index(maxa.kind) 

1015 index_sc = _kind_list.index(maxsc.kind) 

1016 except ValueError: 

1017 return None 

1018 

1019 if index_sc > index_a: 

1020 return _find_common_coerce(maxsc, maxa) 

1021 else: 

1022 return maxa
