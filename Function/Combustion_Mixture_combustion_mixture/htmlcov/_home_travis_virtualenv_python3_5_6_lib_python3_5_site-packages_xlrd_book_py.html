1383def unpack_SST_table(datatab, nstrings): 

1384 "Return list of strings" 

1385 datainx = 0 

1386 ndatas = len(datatab) 

1387 data = datatab[0] 

1388 datalen = len(data) 

1389 pos = 8 

1390 strings = [] 

1391 strappend = strings.append 

1392 richtext_runs = {} 

1393 local_unpack = unpack 

1394 local_min = min 

1395 local_BYTES_ORD = BYTES_ORD 

1396 latin_1 = "latin_1" 

1397 for _unused_i in xrange(nstrings): 

1398 nchars = local_unpack('<H', data[pos:pos+2])[0] 

1399 pos += 2 

1400 options = local_BYTES_ORD(data[pos]) 

1401 pos += 1 

1402 rtcount = 0 

1403 phosz = 0 

1404 if options & 0x08: # richtext 

1405 rtcount = local_unpack('<H', data[pos:pos+2])[0] 

1406 pos += 2 

1407 if options & 0x04: # phonetic 

1408 phosz = local_unpack('<i', data[pos:pos+4])[0] 

1409 pos += 4 

1410 accstrg = UNICODE_LITERAL('') 

1411 charsgot = 0 

1412 while 1: 

1413 charsneed = nchars - charsgot 

1414 if options & 0x01: 

1415 # Uncompressed UTF-16 

1416 charsavail = local_min((datalen - pos) >> 1, charsneed) 

1417 rawstrg = data[pos:pos+2*charsavail] 

1418 # if DEBUG: print "SST U16: nchars=%d pos=%d rawstrg=%r" % (nchars, pos, rawstrg) 

1419 try: 

1420 accstrg += unicode(rawstrg, "utf_16_le") 

1421 except: 

1422 # print "SST U16: nchars=%d pos=%d rawstrg=%r" % (nchars, pos, rawstrg) 

1423 # Probable cause: dodgy data e.g. unfinished surrogate pair. 

1424 # E.g. file unicode2.xls in pyExcelerator's examples has cells containing 

1425 # unichr(i) for i in range(0x100000) 

1426 # so this will include 0xD800 etc 

1427 raise 

1428 pos += 2*charsavail 

1429 else: 

1430 # Note: this is COMPRESSED (not ASCII!) encoding!!! 

1431 charsavail = local_min(datalen - pos, charsneed) 

1432 rawstrg = data[pos:pos+charsavail] 

1433 # if DEBUG: print "SST CMPRSD: nchars=%d pos=%d rawstrg=%r" % (nchars, pos, rawstrg) 

1434 accstrg += unicode(rawstrg, latin_1) 

1435 pos += charsavail 

1436 charsgot += charsavail 

1437 if charsgot == nchars: 

1438 break 

1439 datainx += 1 

1440 data = datatab[datainx] 

1441 datalen = len(data) 

1442 options = local_BYTES_ORD(data[0]) 

1443 pos = 1 

1444 

1445 if rtcount: 

1446 runs = [] 

1447 for runindex in xrange(rtcount): 

1448 if pos == datalen: 

1449 pos = 0 

1450 datainx += 1 

1451 data = datatab[datainx] 

1452 datalen = len(data) 

1453 runs.append(local_unpack("<HH", data[pos:pos+4])) 

1454 pos += 4 

1455 richtext_runs[len(strings)] = runs 

1456 

1457 pos += phosz # size of the phonetic stuff to skip 

1458 if pos >= datalen: 

1459 # adjust to correct position in next record 

1460 pos = pos - datalen 

1461 datainx += 1 

1462 if datainx < ndatas: 

1463 data = datatab[datainx] 

1464 datalen = len(data) 

1465 else: 

1466 assert _unused_i == nstrings - 1 

1467 strappend(accstrg) 

1468 return strings, richtext_runs
