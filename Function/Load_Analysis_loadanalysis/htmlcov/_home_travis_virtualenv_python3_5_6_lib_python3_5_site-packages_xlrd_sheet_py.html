1728 def handle_hlink(self, data): 

1729 # DEBUG = 1 

1730 if DEBUG: print("\n=== hyperlink ===", file=self.logfile) 

1731 record_size = len(data) 

1732 h = Hyperlink() 

1733 h.frowx, h.lrowx, h.fcolx, h.lcolx, guid0, dummy, options = unpack('<HHHH16s4si', data[:32]) 

1734 assert guid0 == b"\xD0\xC9\xEA\x79\xF9\xBA\xCE\x11\x8C\x82\x00\xAA\x00\x4B\xA9\x0B" 

1735 assert dummy == b"\x02\x00\x00\x00"

1739 def get_nul_terminated_unicode(buf, ofs): 

1740 nb = unpack('<L', buf[ofs:ofs+4])[0] * 2 

1741 ofs += 4 

1742 uc = unicode(buf[ofs:ofs+nb], 'UTF-16le')[:-1] 

1743 ofs += nb 

1744 return uc, ofs 

1745 

1746 if options & 0x14: # has a description 

1747 h.desc, offset = get_nul_terminated_unicode(data, offset) 

1748 

1749 if options & 0x80: # has a target 

1750 h.target, offset = get_nul_terminated_unicode(data, offset) 

1751 

1752 if (options & 1) and not (options & 0x100): # HasMoniker and not MonikerSavedAsString 

1753 # an OLEMoniker structure 

1754 clsid, = unpack('<16s', data[offset:offset + 16]) 

1755 if DEBUG: fprintf(self.logfile, "clsid=%r\n", clsid) 

1756 offset += 16

1758 # E0H C9H EAH 79H F9H BAH CEH 11H 8CH 82H 00H AAH 00H 4BH A9H 0BH 

1759 # URL Moniker 

1760 h.type = UNICODE_LITERAL('url') 

1761 nbytes = unpack('<L', data[offset:offset + 4])[0] 

1762 offset += 4 

1763 h.url_or_path = unicode(data[offset:offset + nbytes], 'UTF-16le') 

1764 if DEBUG: fprintf(self.logfile, "initial url=%r len=%d\n", h.url_or_path, len(h.url_or_path))

1767 h.url_or_path = h.url_or_path[:endpos]

1769 offset += true_nbytes 

1770 extra_nbytes = nbytes - true_nbytes 

1771 extra_data = data[offset:offset + extra_nbytes] 

1772 offset += extra_nbytes 

1773 if DEBUG: 

1774 fprintf( 

1775 self.logfile, 

1776 "url=%r\nextra=%r\nnbytes=%d true_nbytes=%d extra_nbytes=%d\n", 

1777 h.url_or_path, extra_data, nbytes, true_nbytes, extra_nbytes, 

1778 ) 

1779 assert extra_nbytes in (24, 0) 

1780 elif clsid == b"\x03\x03\x00\x00\x00\x00\x00\x00\xC0\x00\x00\x00\x00\x00\x00\x46":

1782 h.type = UNICODE_LITERAL('local file')

1784 offset += 6 

1785 shortpath = b"..\\" * uplevels + data[offset:offset + nbytes - 1] #### BYTES, not unicode 

1786 if DEBUG: fprintf(self.logfile, "uplevels=%d shortpath=%r\n", uplevels, shortpath) 

1787 offset += nbytes 

1788 offset += 24 # OOo: "unknown byte sequence" 

1789 # above is version 0xDEAD + 20 reserved zero bytes 

1790 sz = unpack('<i', data[offset:offset + 4])[0]

1792 offset += 4 

1793 if sz: 

1794 xl = unpack('<i', data[offset:offset + 4])[0] 

1795 offset += 4 

1796 offset += 2 # "unknown byte sequence" MS: 0x0003 

1797 extended_path = unicode(data[offset:offset + xl], 'UTF-16le') # not zero-terminated 

1798 offset += xl 

1799 h.url_or_path = extended_path 

1800 else: 

1801 h.url_or_path = shortpath 

1802 #### MS KLUDGE WARNING #### 

1803 # The "shortpath" is bytes encoded in the **UNKNOWN** creator's "ANSI" encoding. 

1804 else: 

1805 fprintf(self.logfile, "*** unknown clsid %r\n", clsid) 

1806 elif options & 0x163 == 0x103: # UNC 

1807 h.type = UNICODE_LITERAL('unc') 

1808 h.url_or_path, offset = get_nul_terminated_unicode(data, offset) 

1809 elif options & 0x16B == 8: 

1810 h.type = UNICODE_LITERAL('workbook') 

1811 else: 

1812 h.type = UNICODE_LITERAL('unknown') 

1813 

1814 if options & 0x8: # has textmark 

1815 h.textmark, offset = get_nul_terminated_unicode(data, offset) 

1816 

1817 if DEBUG: 

1818 h.dump(header="... object dump ...") 

1819 print("offset=%d record_size=%d" % (offset, record_size)) 

1820 

1821 extra_nbytes = record_size - offset 

1822 if extra_nbytes > 0: 

1823 fprintf( 

1824 self.logfile, 

1825 "*** WARNING: hyperlink at R%dC%d has %d extra data bytes: %s\n", 

1826 h.frowx + 1, 

1827 h.fcolx + 1,

1830 )

1837 for colx in xrange(h.fcolx, h.lcolx+1): 

1838 self.hyperlink_map[rowx, colx] = h
