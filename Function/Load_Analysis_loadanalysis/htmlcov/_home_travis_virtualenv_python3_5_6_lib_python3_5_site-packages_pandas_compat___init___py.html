30def set_function_name(f, name, cls): 

31 """ 

32 Bind the name/qualname attributes of the function 

33 """ 

34 f.__name__ = name 

35 f.__qualname__ = "{klass}.{name}".format(klass=cls.__name__, name=name) 

36 f.__module__ = cls.__module__ 

37 return f

40def raise_with_traceback(exc, traceback=Ellipsis): 

41 """ 

42 Raise exception with existing traceback. 

43 If traceback is not passed, uses sys.exc_info() to get traceback. 

44 """ 

45 if traceback == Ellipsis: 

46 _, _, traceback = sys.exc_info() 

47 raise exc.with_traceback(traceback)
