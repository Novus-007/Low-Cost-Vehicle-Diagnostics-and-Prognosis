38def figure_edit(axes, parent=None):

42 # Get / General 

43 # Cast to builtin floats as they have nicer reprs. 

44 xmin, xmax = map(float, axes.get_xlim()) 

45 ymin, ymax = map(float, axes.get_ylim()) 

46 general = [('Title', axes.get_title()), 

47 sep, 

48 (None, "<b>X-Axis</b>"), 

49 ('Left', xmin), ('Right', xmax), 

50 ('Label', axes.get_xlabel()), 

51 ('Scale', [axes.get_xscale(), 'linear', 'log', 'logit']), 

52 sep, 

53 (None, "<b>Y-Axis</b>"), 

54 ('Bottom', ymin), ('Top', ymax), 

55 ('Label', axes.get_ylabel()), 

56 ('Scale', [axes.get_yscale(), 'linear', 'log', 'logit']), 

57 sep, 

58 ('(Re-)Generate automatic legend', False), 

59 ]

61 # Save the unit data 

62 xconverter = axes.xaxis.converter 

63 yconverter = axes.yaxis.converter 

64 xunits = axes.xaxis.get_units() 

65 yunits = axes.yaxis.get_units()

67 # Sorting for default labels (_lineXXX, _imageXXX). 

68 def cmp_key(label): 

69 match = re.match(r"(_line|_image)(\d+)", label) 

70 if match: 

71 return match.group(1), int(match.group(2)) 

72 else: 

73 return label, 0

77 for line in axes.get_lines(): 

78 label = line.get_label() 

79 if label == '_nolegend_':

84 def prepare_data(d, init): 

85 """Prepare entry for FormLayout. 

86 

87 `d` is a mapping of shorthands to style names (a single style may 

88 have multiple shorthands, in particular the shorthands `None`, 

89 `"None"`, `"none"` and `""` are synonyms); `init` is one shorthand 

90 of the initial style. 

91 

92 This function returns an list suitable for initializing a 

93 FormLayout combobox, namely `[initial_name, (shorthand, 

94 style_name), (shorthand, style_name), ...]`. 

95 """ 

96 if init not in d: 

97 d = {**d, init: str(init)} 

98 # Drop duplicate shorthands from dict (by overwriting them during 

99 # the dict comprehension). 

100 name2short = {name: short for short, name in d.items()} 

101 # Convert back to {shorthand: name}. 

102 short2name = {short: name for name, short in name2short.items()} 

103 # Find the kept shorthand for the style specified by init. 

104 canonical_init = name2short[d[init]] 

105 # Sort by representation and prepend the initial value. 

106 return ([canonical_init] + 

107 sorted(short2name.items(), 

108 key=lambda short_and_name: short_and_name[1])) 

109 

110 curvelabels = sorted(linedict, key=cmp_key) 

111 for label in curvelabels: 

112 line = linedict[label] 

113 color = mcolors.to_hex( 

114 mcolors.to_rgba(line.get_color(), line.get_alpha()), 

115 keep_alpha=True) 

116 ec = mcolors.to_hex( 

117 mcolors.to_rgba(line.get_markeredgecolor(), line.get_alpha()), 

118 keep_alpha=True) 

119 fc = mcolors.to_hex( 

120 mcolors.to_rgba(line.get_markerfacecolor(), line.get_alpha()), 

121 keep_alpha=True) 

122 curvedata = [ 

123 ('Label', label), 

124 sep, 

125 (None, '<b>Line</b>'), 

126 ('Line style', prepare_data(LINESTYLES, line.get_linestyle())), 

127 ('Draw style', prepare_data(DRAWSTYLES, line.get_drawstyle())), 

128 ('Width', line.get_linewidth()), 

129 ('Color (RGBA)', color), 

130 sep, 

131 (None, '<b>Marker</b>'), 

132 ('Style', prepare_data(MARKERS, line.get_marker())), 

133 ('Size', line.get_markersize()), 

134 ('Face color (RGBA)', fc), 

135 ('Edge color (RGBA)', ec)] 

136 curves.append([curvedata, label, ""]) 

137 # Is there a curve displayed? 

138 has_curve = bool(curves) 

139 

140 # Get / Images 

141 imagedict = {} 

142 for image in axes.get_images(): 

143 label = image.get_label() 

144 if label == '_nolegend_': 

145 continue 

146 imagedict[label] = image 

147 imagelabels = sorted(imagedict, key=cmp_key) 

148 images = [] 

149 cmaps = [(cmap, name) for name, cmap in sorted(cm.cmap_d.items())] 

150 for label in imagelabels: 

151 image = imagedict[label] 

152 cmap = image.get_cmap() 

153 if cmap not in cm.cmap_d.values(): 

154 cmaps = [(cmap, cmap.name)] + cmaps 

155 low, high = image.get_clim() 

156 imagedata = [ 

157 ('Label', label), 

158 ('Colormap', [cmap.name] + cmaps), 

159 ('Min. value', low), 

160 ('Max. value', high), 

161 ('Interpolation', 

162 [image.get_interpolation()] 

163 + [(name, name) for name in sorted(mimage.interpolations_names)])] 

164 images.append([imagedata, label, ""]) 

165 # Is there an image displayed? 

166 has_image = bool(images) 

167 

168 datalist = [(general, "Axes", "")] 

169 if curves: 

170 datalist.append((curves, "Curves", "")) 

171 if images: 

172 datalist.append((images, "Images", ""))

174 def apply_callback(data): 

175 """This function will be called to apply changes""" 

176 orig_xlim = axes.get_xlim() 

177 orig_ylim = axes.get_ylim() 

178 

179 general = data.pop(0) 

180 curves = data.pop(0) if has_curve else [] 

181 images = data.pop(0) if has_image else [] 

182 if data: 

183 raise ValueError("Unexpected field") 

184 

185 # Set / General 

186 (title, xmin, xmax, xlabel, xscale, ymin, ymax, ylabel, yscale, 

187 generate_legend) = general 

188 

189 if axes.get_xscale() != xscale: 

190 axes.set_xscale(xscale) 

191 if axes.get_yscale() != yscale: 

192 axes.set_yscale(yscale) 

193 

194 axes.set_title(title) 

195 axes.set_xlim(xmin, xmax) 

196 axes.set_xlabel(xlabel) 

197 axes.set_ylim(ymin, ymax) 

198 axes.set_ylabel(ylabel) 

199 

200 # Restore the unit data 

201 axes.xaxis.converter = xconverter 

202 axes.yaxis.converter = yconverter 

203 axes.xaxis.set_units(xunits) 

204 axes.yaxis.set_units(yunits) 

205 axes.xaxis._update_axisinfo() 

206 axes.yaxis._update_axisinfo() 

207 

208 # Set / Curves 

209 for index, curve in enumerate(curves): 

210 line = linedict[curvelabels[index]] 

211 (label, linestyle, drawstyle, linewidth, color, marker, markersize, 

212 markerfacecolor, markeredgecolor) = curve 

213 line.set_label(label) 

214 line.set_linestyle(linestyle) 

215 line.set_drawstyle(drawstyle) 

216 line.set_linewidth(linewidth) 

217 rgba = mcolors.to_rgba(color) 

218 line.set_alpha(None) 

219 line.set_color(rgba) 

220 if not isinstance(marker, str) or marker != 'none': 

221 line.set_marker(marker) 

222 line.set_markersize(markersize) 

223 line.set_markerfacecolor(markerfacecolor) 

224 line.set_markeredgecolor(markeredgecolor) 

225 

226 # Set / Images 

227 for index, image_settings in enumerate(images): 

228 image = imagedict[imagelabels[index]] 

229 label, cmap, low, high, interpolation = image_settings 

230 image.set_label(label) 

231 image.set_cmap(cm.get_cmap(cmap)) 

232 image.set_clim(*sorted([low, high])) 

233 image.set_interpolation(interpolation) 

234 

235 # re-generate legend, if checkbox is checked 

236 if generate_legend: 

237 draggable = None 

238 ncol = 1 

239 if axes.legend_ is not None: 

240 old_legend = axes.get_legend() 

241 draggable = old_legend._draggable is not None 

242 ncol = old_legend._ncol 

243 new_legend = axes.legend(ncol=ncol) 

244 if new_legend: 

245 new_legend.set_draggable(draggable) 

246 

247 # Redraw 

248 figure = axes.get_figure() 

249 figure.canvas.draw() 

250 if not (axes.get_xlim() == orig_xlim and axes.get_ylim() == orig_ylim): 

251 figure.canvas.toolbar.push_current() 

252 

253 data = formlayout.fedit(datalist, title="Figure options", parent=parent, 

254 icon=get_icon('qt4_editor_options.svg'), 

255 apply=apply_callback) 

256 if data is not None: 

257 apply_callback(data)
