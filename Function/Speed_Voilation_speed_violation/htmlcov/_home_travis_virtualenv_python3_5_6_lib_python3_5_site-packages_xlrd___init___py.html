33def open_workbook(filename=None, 

34 logfile=sys.stdout, 

35 verbosity=0, 

36 use_mmap=USE_MMAP, 

37 file_contents=None, 

38 encoding_override=None, 

39 formatting_info=False, 

40 on_demand=False, 

41 ragged_rows=False): 

42 """ 

43 Open a spreadsheet file for data extraction. 

44 

45 :param filename: The path to the spreadsheet file to be opened. 

46 

47 :param logfile: An open file to which messages and diagnostics are written. 

48 

49 :param verbosity: Increases the volume of trace material written to the 

50 logfile. 

51 

52 :param use_mmap: 

53 

54 Whether to use the mmap module is determined heuristically. 

55 Use this arg to override the result. 

56 

57 Current heuristic: mmap is used if it exists. 

58 

59 :param file_contents: 

60 

61 A string or an :class:`mmap.mmap` object or some other behave-alike 

62 object. If ``file_contents`` is supplied, ``filename`` will not be used, 

63 except (possibly) in messages. 

64 

65 :param encoding_override: 

66 

67 Used to overcome missing or bad codepage information 

68 in older-version files. See :doc:`unicode`. 

69 

70 :param formatting_info: 

71 

72 The default is ``False``, which saves memory. 

73 In this case, "Blank" cells, which are those with their own formatting 

74 information but no data, are treated as empty by ignoring the file's 

75 ``BLANK`` and ``MULBLANK`` records. 

76 This cuts off any bottom or right "margin" of rows of empty or blank 

77 cells. 

78 Only :meth:`~xlrd.sheet.Sheet.cell_value` and 

79 :meth:`~xlrd.sheet.Sheet.cell_type` are available. 

80 

81 When ``True``, formatting information will be read from the spreadsheet 

82 file. This provides all cells, including empty and blank cells. 

83 Formatting information is available for each cell. 

84 

85 Note that this will raise a NotImplementedError when used with an 

86 xlsx file. 

87 

88 :param on_demand: 

89 

90 Governs whether sheets are all loaded initially or when demanded 

91 by the caller. See :doc:`on_demand`. 

92 

93 :param ragged_rows: 

94 

95 The default of ``False`` means all rows are padded out with empty cells so 

96 that all rows have the same size as found in 

97 :attr:`~xlrd.sheet.Sheet.ncols`. 

98 

99 ``True`` means that there are no empty cells at the ends of rows. 

100 This can result in substantial memory savings if rows are of widely 

101 varying sizes. See also the :meth:`~xlrd.sheet.Sheet.row_len` method. 

102 

103 :returns: An instance of the :class:`~xlrd.book.Book` class. 

104 """ 

105 

106 peeksz = 4 

107 if file_contents: 

108 peek = file_contents[:peeksz] 

109 else: 

110 filename = os.path.expanduser(filename) 

111 with open(filename, "rb") as f: 

112 peek = f.read(peeksz) 

113 if peek == b"PK\x03\x04": # a ZIP file 

114 if file_contents: 

115 zf = zipfile.ZipFile(timemachine.BYTES_IO(file_contents)) 

116 else: 

117 zf = zipfile.ZipFile(filename) 

118 

119 # Workaround for some third party files that use forward slashes and 

120 # lower case names. We map the expected name in lowercase to the 

121 # actual filename in the zip container. 

122 component_names = dict([(X12Book.convert_filename(name), name) 

123 for name in zf.namelist()]) 

124 

125 if verbosity: 

126 logfile.write('ZIP component_names:\n') 

127 pprint.pprint(component_names, logfile) 

128 if 'xl/workbook.xml' in component_names: 

129 from . import xlsx 

130 bk = xlsx.open_workbook_2007_xml( 

131 zf, 

132 component_names, 

133 logfile=logfile, 

134 verbosity=verbosity, 

135 use_mmap=use_mmap, 

136 formatting_info=formatting_info, 

137 on_demand=on_demand, 

138 ragged_rows=ragged_rows, 

139 ) 

140 return bk 

141 if 'xl/workbook.bin' in component_names: 

142 raise XLRDError('Excel 2007 xlsb file; not supported') 

143 if 'content.xml' in component_names: 

144 raise XLRDError('Openoffice.org ODS file; not supported') 

145 raise XLRDError('ZIP file contents not a known type of workbook') 

146 

147 from . import book 

148 bk = book.open_workbook_xls( 

149 filename=filename, 

150 logfile=logfile, 

151 verbosity=verbosity, 

152 use_mmap=use_mmap, 

153 file_contents=file_contents, 

154 encoding_override=encoding_override, 

155 formatting_info=formatting_info, 

156 on_demand=on_demand, 

157 ragged_rows=ragged_rows, 

158 ) 

159 return bk
