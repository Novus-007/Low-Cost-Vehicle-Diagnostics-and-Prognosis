42def _remove_blacklisted_style_params(d, warn=True): 

43 o = {} 

44 for key, val in d.items(): 

45 if key in STYLE_BLACKLIST: 

46 if warn: 

47 warnings.warn( 

48 "Style includes a parameter, '{0}', that is not related " 

49 "to style. Ignoring".format(key), stacklevel=3) 

50 else: 

51 o[key] = val 

52 return o

64def use(style): 

65 """Use matplotlib style settings from a style specification. 

66 

67 The style name of 'default' is reserved for reverting back to 

68 the default style settings. 

69 

70 Parameters 

71 ---------- 

72 style : str, dict, or list 

73 A style specification. Valid options are: 

74 

75 +------+-------------------------------------------------------------+ 

76 | str | The name of a style or a path/URL to a style file. For a | 

77 | | list of available style names, see `style.available`. | 

78 +------+-------------------------------------------------------------+ 

79 | dict | Dictionary with valid key/value pairs for | 

80 | | `matplotlib.rcParams`. | 

81 +------+-------------------------------------------------------------+ 

82 | list | A list of style specifiers (str or dict) applied from first | 

83 | | to last in the list. | 

84 +------+-------------------------------------------------------------+ 

85 

86 

87 """ 

88 style_alias = {'mpl20': 'default', 

89 'mpl15': 'classic'} 

90 if isinstance(style, str) or hasattr(style, 'keys'): 

91 # If name is a single str or dict, make it a single element list. 

92 styles = [style] 

93 else: 

94 styles = style 

95 

96 styles = (style_alias.get(s, s) if isinstance(s, str) else s 

97 for s in styles) 

98 for style in styles: 

99 if not isinstance(style, str): 

100 _apply_style(style) 

101 elif style == 'default': 

102 # Deprecation warnings were already handled when creating 

103 # rcParamsDefault, no need to reemit them here. 

104 with warnings.catch_warnings(): 

105 warnings.simplefilter("ignore", MatplotlibDeprecationWarning) 

106 _apply_style(rcParamsDefault, warn=False) 

107 elif style in library: 

108 _apply_style(library[style]) 

109 else: 

110 try: 

111 rc = rc_params_from_file(style, use_default_template=False) 

112 _apply_style(rc) 

113 except IOError: 

114 raise IOError( 

115 "{!r} not found in the style library and input is not a " 

116 "valid URL or path; see `style.available` for list of " 

117 "available styles".format(style))

120@contextlib.contextmanager 

121def context(style, after_reset=False): 

122 """Context manager for using style settings temporarily. 

123 

124 Parameters 

125 ---------- 

126 style : str, dict, or list 

127 A style specification. Valid options are: 

128 

129 +------+-------------------------------------------------------------+ 

130 | str | The name of a style or a path/URL to a style file. For a | 

131 | | list of available style names, see `style.available`. | 

132 +------+-------------------------------------------------------------+ 

133 | dict | Dictionary with valid key/value pairs for | 

134 | | `matplotlib.rcParams`. | 

135 +------+-------------------------------------------------------------+ 

136 | list | A list of style specifiers (str or dict) applied from first | 

137 | | to last in the list. | 

138 +------+-------------------------------------------------------------+ 

139 

140 after_reset : bool 

141 If True, apply style after resetting settings to their defaults; 

142 otherwise, apply style on top of the current settings. 

143 """ 

144 with mpl.rc_context(): 

145 if after_reset: 

146 mpl.rcdefaults() 

147 use(style) 

148 yield

157def iter_user_libraries(): 

158 for stylelib_path in USER_LIBRARY_PATHS:

164def update_user_library(library): 

165 """Update style library with user-defined rc files""" 

166 for stylelib_path in iter_user_libraries(): 

167 styles = read_style_directory(stylelib_path) 

168 update_nested_dict(library, styles)

172def iter_style_files(style_dir): 

173 """Yield file path and name of styles in the given directory.""" 

174 for path in os.listdir(style_dir): 

175 filename = os.path.basename(path) 

176 if is_style_file(filename): 

177 match = STYLE_FILE_PATTERN.match(filename) 

178 path = os.path.abspath(os.path.join(style_dir, path)) 

179 yield path, match.groups()[0]

182def read_style_directory(style_dir): 

183 """Return dictionary of styles defined in `style_dir`.""" 

184 styles = dict() 

185 for path, name in iter_style_files(style_dir): 

186 with warnings.catch_warnings(record=True) as warns: 

187 styles[name] = rc_params_from_file(path, 

188 use_default_template=False) 

189 

190 for w in warns: 

191 message = 'In %s: %s' % (path, w.message) 

192 warnings.warn(message, stacklevel=2) 

193 

194 return styles

197def update_nested_dict(main_dict, new_dict): 

198 """Update nested dict (only level of nesting) with new values. 

199 

200 Unlike dict.update, this assumes that the values of the parent dict are 

201 dicts (or dict-like), so you shouldn't replace the nested dict if it 

202 already exists. Instead you should update the sub-dict. 

203 """ 

204 # update named styles specified by user 

205 for name, rc_dict in new_dict.items(): 

206 main_dict.setdefault(name, {}).update(rc_dict) 

207 return main_dict
